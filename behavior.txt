1. all balls move according to gravity
2. all balls move according to ball collision
3. all balls move according to wall collision
This can all be handled by the ball spawners

        for ball in balls:
            move each ball according to gravity

    
    def move_balls_by_ball_collision():
        balls_want_to_move_like_this = [(0, 0) for i in range(len(balls))]
        collisions = [0 for in range(len(balls))]

        repulsion_damper = 2
        for i in range(len(balls)):
            for j in range(len(balls)):
                if i == j:
                    pass

                b1 = balls[i]
                b2 = balls[j]

                if b1 and b2 are overlapping
                    collisions[i] += 1

                    move_by_X = b1.centerX - b2.centerX
                    move_by_Y = b1.centerY - b2.centerY
                    if move_by_X < 0:
                        rX = -b1.radius
                    else:
                        rX = b1.radius
                    if move_by_Y < 0:
                        rY = -b1.radius
                    else:
                        rY = b1.radius
                    move_by_X = (move_by_X + rX) / repulsion_damper
                    move_by_Y = (move_by_Y + rY) / repulsion_damper

                    balls_want_to_move_like_this[i][0] += move_by_X
                    balls_want_to_move_like_this[i][1] += move_by_Y

        for i in range(len(balls_want_to_move_like_this)):
            if collisions[i] > 0:
                balls_want_to_move_like_this[i][0] /= collisions[i]
                balls_want_to_move_like_this[i][1] /= collisions[i]
